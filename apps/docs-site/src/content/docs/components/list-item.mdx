A row primitive for lists. Title plus optional subtitle, with `left` and `right` slots that accept any `ReactNode` — icons, avatars, badges, switches, chevrons.

<ComponentPreview slug="components/list-item" />

## Install

```sh
npx stylesheet-ui add list-item
```

## Usage

```tsx
import { ListItem } from "@/components/ui/list-item";
import { Avatar } from "@/components/ui/avatar";
import { Badge } from "@/components/ui/badge";
import { ChevronRight } from "lucide-react-native";

<ListItem
  title="Inbox"
  subtitle="4 unread messages"
  right={<Badge size="sm">4</Badge>}
/>

<ListItem
  title="Ada Lovelace"
  subtitle="ada@example.com"
  left={<Avatar size="md" name="Ada Lovelace" />}
  right={<ChevronRight size={18} />}
  onPress={() => {}}
/>
```

## Props

| Prop       | Type                                | Default | Description |
|------------|-------------------------------------|---------|-------------|
| `title`    | `string`                            | —       | Primary label |
| `subtitle` | `string`                            | —       | Secondary line below the title |
| `left`     | `ReactNode`                         | —       | Element rendered before the title |
| `right`    | `ReactNode`                         | —       | Element rendered after the title |
| `onPress`  | `PressableProps["onPress"]`         | —       | Makes the row pressable when provided |
| `disabled` | `boolean`                           | `false` | Mutes opacity; disables press |
